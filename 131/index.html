2012-07-02 15:30 #git 

alias gg="git grep -H --heading --break"

[grep]
  lineNumber = true

  • -Hでファイル名を相対パスにする
  • --headingでファイル名をヒットした行の上にまとめて表示する
  • --breakで空行を入れる
  • git config --global grep.lineNumber trueと打つと、デフォルトでヒットした行数を表示する
