set nocompatible

" ----------------------------------------
" Colors
" ----------------------------------------
colorscheme railscasts

syntax on

highlight StatusLine   cterm=NONE ctermfg=white ctermbg=233
highlight StatusLineNC cterm=NONE ctermfg=white ctermbg=233
highlight VertSplit    cterm=NONE ctermfg=233   ctermbg=233

" ----------------------------------------
" Display
" ----------------------------------------
set laststatus=2
set number
set ruler
set showcmd
set showmatch
set list
set listchars=tab:->,trail:-

" ----------------------------------------
" Edit
" ----------------------------------------
set autoindent smartindent
set expandtab
set tabstop=2 softtabstop=0 shiftwidth=2
set wildmenu
set wildmode=list:full

" ----------------------------------------
" Search
" ----------------------------------------
set hlsearch
set incsearch

nnoremap n nzz 
nnoremap N Nzz 
nnoremap  :nohlsearch

" ----------------------------------------
" Vundle
" ----------------------------------------
filetype off 

set rtp+=~/.vim/bundle/vundle/
call vundle#rc()

" manage plugins
Bundle 'gmarik/vundle'

" search files
Bundle 'unite.vim'

" complete keys
Bundle 'Shougo/neocomplcache'

" complete snippets
Bundle 'Shougo/neocomplcache-snippets-complete'

" git command on vim
Bundle 'fugitive.vim'

" format codes
Bundle 'Align'

" toggle comment outs
Bundle 'git://github.com/scrooloose/nerdcommenter.git'

" HTML snippets
Bundle 'ZenCoding.vim'

" tree explorer
Bundle 'git://github.com/scrooloose/nerdtree.git'

" bettar indentation for JavaScript
Bundle 'git://github.com/pangloss/vim-javascript.git'

" syntax and indent for coffee script
Bundle 'kchmck/vim-coffee-script'

" plugins for Titanium Desktop/Mobile
Bundle 'git://github.com/pekepeke/titanium-vim.git'

filetype plugin indent on

" ----------------------------------------
" Unite.vim
" ----------------------------------------
let g:unite_enable_start_insert = 1

" open unite buffer
nnoremap  :Unite buffer file_rec

" ----------------------------------------
" neocomplcache.vim
" ----------------------------------------
let g:neocomplcache_enable_at_startup = 1

highlight Pmenu     ctermbg=8
highlight PmenuSel  ctermbg=1
highlight PmenuSbar ctermbg=0

imap  neocomplcache#sources#snippets_complete#expandable() ? "\(neocomplcache_snippets_expand)" : pumvisible() ? "\" : "\"
smap  neocomplcache#sources#snippets_complete#expandable() ? "\(neocomplcache_snippets_expand)" : pumvisible() ? "\" : "\"

inoremap   neocomplcache#smart_close_popup()."\"
inoremap  neocomplcache#close_popup()
inoremap  neocomplcache#cancel_popup()

" ----------------------------------------
" Align.vim
" ----------------------------------------
let g:Align_xstrlen = 3

" ----------------------------------------
" NERD_commenter.vim
" ----------------------------------------
let g:NERDSpaceDelims = 1
let g:NERDShutup = 1

map x NERDCommenterToggle

" ----------------------------------------
" Other Key Mappings
" ----------------------------------------

" reload .vimrc
nnoremap r :source ~/.vimrc

" move by one display line
nnoremap j gj
nnoremap k gk

" move to line head/end like mac
nnoremap  0
nnoremap  $

" go to another window
nnoremap h h
nnoremap j j
nnoremap k k
nnoremap l l

" move the window to another position
nnoremap H H
nnoremap J J
nnoremap K K
nnoremap L L

" adjust the window height/width
nnoremap + 5+
nnoremap - 5-
nnoremap > 10>
nnoremap < 10<

" complete surroundings
inoremap ' ''
inoremap " ""
inoremap ( ()
inoremap { {}
inoremap [ []
inoremap \| \|\|

" hjkl move in insert mode
inoremap  
inoremap  
inoremap  
inoremap  

" ctags
nnoremap  :pop
