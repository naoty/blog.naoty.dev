2012-01-17 23:01 #ruby 

namespace :logging do
  task :tail_unicorn do
    run "tail -f #{shared_path}/log/unicorn.log" do |channel, stream, data|
      trap("INT") { puts "Interupted!"; exit 0; }
      puts "=== #{channel[:host]} ==="
      puts data
      break if stream == :err
    end
  end
end
